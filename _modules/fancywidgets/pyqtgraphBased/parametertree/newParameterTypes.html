[docs]    def makeWidget(self):
        v = self.param.opts.get('value',self.param.opts.get('limits',[''])[0])
        w = QtGui.QMenuBar()
        w.menu=w.addMenu(v)
        w.sigChanged = None
        w.value = lambda: ''
        w.menu.aboutToShow.connect(lambda menu=w.menu:self.param.aboutToShow.emit(menu))
        return w
