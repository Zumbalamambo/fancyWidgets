[docs]	def blockSignals(self, boolean):
		'''add block/unblock of keyboard shortcut'''
		pgParameter.blockSignals(self, boolean)
		try:
			item = self.items[0]
			if item.key:
				item.key.blockSignals(boolean)
		except:
			pass

[docs]	def isVisible(self):
		if self.opts['visible']:
			p = self
			while True:
				p = p.parent()
				if not p:
					return True
				if not p.opts['visible']:
					break
		return False

[docs]	def path(self):
		c = p = self.parent()
		l = self.name()
		while p:
			l = p.name() + ', ' + l
			c = p
			p = p.parent()
		return c, l
